function spinWords(string) {
        let array = string.split(" ");
        let str = "";

        for (let word of array) {
          str += " ";
          if (word.length >= 5) {
            str += word.split("").reverse().join("");
          } else {
            str += word;
          }
        }
        return str.trim();
      }
